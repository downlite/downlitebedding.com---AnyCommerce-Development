# remove legacy cart and session variables from uri
~path	/c=*/(.*?)				 /$1			set=path
~path	/s=*/(.*?)				/$1			set=path

# redirect all product and category urls
~path	/category/([^/]*)	/#!category/$1
~path	/product/([^/]*)	/#!product/$1
